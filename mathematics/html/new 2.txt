    <script>
        let selectedFile = null;
        let extractedHTML = null;
        let extractedFiles = {};

        // Drag & drop handlers
        const uploadArea = document.getElementById('uploadArea');
        const fileInput = document.getElementById('fileInput');
        const fileInfo = document.getElementById('fileInfo');
        const fileName = document.getElementById('fileName');
        const fileSize = document.getElementById('fileSize');
        const convertBtn = document.getElementById('convertBtn');
        const progress = document.getElementById('progress');
        const progressFill = document.getElementById('progressFill');
        const progressText = document.getElementById('progressText');
        const resultDiv = document.getElementById('result');
        const preview = document.getElementById('preview');
        const previewFrame = document.getElementById('previewFrame');

        uploadArea.addEventListener('click', () => fileInput.click());
        
        uploadArea.addEventListener('dragover', (e) => {
            e.preventDefault();
            uploadArea.classList.add('drag-over');
        });

        uploadArea.addEventListener('dragleave', () => {
            uploadArea.classList.remove('drag-over');
        });

        uploadArea.addEventListener('drop', (e) => {
            e.preventDefault();
            uploadArea.classList.remove('drag-over');
            const files = e.dataTransfer.files;
            if (files.length > 0) {
                handleFile(files[0]);
            }
        });

        fileInput.addEventListener('change', (e) => {
            if (e.target.files.length > 0) {
                handleFile(e.target.files[0]);
            }
        });

        function handleFile(file) {
            if (!file.name.endsWith('.zip') && !file.name.endsWith('.html')) {
                showResult('Please upload a ZIP file or HTML file', 'error');
                return;
            }
            
            selectedFile = file;
            fileName.textContent = file.name;
            fileSize.textContent = ` (${(file.size / 1024).toFixed(2)} KB)`;
            fileInfo.classList.add('show');
            convertBtn.disabled = false;
            hideResult();
            hidePreview();
        }

        function clearFile() {
            selectedFile = null;
            extractedHTML = null;
            extractedFiles = {};
            fileInfo.classList.remove('show');
            convertBtn.disabled = true;
            hideResult();
            hidePreview();
            fileInput.value = '';
        }

        async function convertToPDF() {
            if (!selectedFile) return;
            
            convertBtn.disabled = true;
            progress.classList.add('show');
            progressFill.style.width = '20%';
            progressText.textContent = 'Reading file...';
            
            try {
                if (selectedFile.name.endsWith('.zip')) {
                    await processZIP(selectedFile);
                } else if (selectedFile.name.endsWith('.html')) {
                    await processHTML(selectedFile);
                }
            } catch (error) {
                console.error(error);
                showResult(`Error: ${error.message}`, 'error');
                progress.classList.remove('show');
                convertBtn.disabled = false;
            }
        }

        async function processZIP(zipFile) {
            progressFill.style.width = '40%';
            progressText.textContent = 'Extracting ZIP file...';
            
            const zip = await JSZip.loadAsync(zipFile);
            extractedFiles = {};
            
            // Extract all files
            const files = Object.keys(zip.files);
            let htmlContent = null;
            let htmlPath = null;
            
            for (const filename of files) {
                const file = zip.files[filename];
                if (!file.dir) {
                    const content = await file.async('blob');
                    extractedFiles[filename] = content;
                    
                    if (filename.endsWith('.html') || filename.endsWith('.htm')) {
                        htmlContent = await content.text();
                        htmlPath = filename;
                    }
                }
            }
            
            if (!htmlContent) {
                throw new Error('No HTML file found in ZIP');
            }
            
            progressFill.style.width = '60%';
            progressText.textContent = 'Processing HTML and images...';
            
            // Process HTML to handle relative paths
            let processedHTML = htmlContent;
            const baseDir = htmlPath.substring(0, htmlPath.lastIndexOf('/') + 1);
            
            // Replace image src paths with blob URLs
            const tempDiv = document.createElement('div');
            tempDiv.innerHTML = htmlContent;
            const images = tempDiv.querySelectorAll('img');
            
            for (const img of images) {
                let src = img.getAttribute('src');
                if (src && !src.startsWith('data:') && !src.startsWith('http')) {
                    // Handle relative paths
                    let fullPath = src;
                    if (src.startsWith('./')) fullPath = src.substring(2);
                    if (src.startsWith('../')) {
                        // Simplify ../ paths
                        fullPath = src.replace(/\.\.\//g, '');
                    }
                    if (baseDir && !fullPath.startsWith('/')) {
                        fullPath = baseDir + fullPath;
                    }
                    
                    const imageBlob = extractedFiles[fullPath];
                    if (imageBlob) {
                        const blobUrl = URL.createObjectURL(imageBlob);
                        img.setAttribute('src', blobUrl);
                    }
                }
            }
            
            processedHTML = tempDiv.innerHTML;
            progressFill.style.width = '80%';
            progressText.textContent = 'Generating PDF...';
            
            await generatePDF(processedHTML);
        }

        async function processHTML(htmlFile) {
            progressFill.style.width = '60%';
            progressText.textContent = 'Reading HTML file...';
            
            const htmlContent = await htmlFile.text();
            
            progressFill.style.width = '80%';
            progressText.textContent = 'Generating PDF...';
            
            await generatePDF(htmlContent);
        }

        async function generatePDF(htmlContent) {
            // Create a temporary container
            const container = document.createElement('div');
            container.innerHTML = htmlContent;
            container.style.padding = '20px';
            container.style.background = 'white';
            
            // Add KaTeX if needed
            if (htmlContent.includes('katex') || htmlContent.includes('\\[')) {
                const katexLink = document.createElement('link');
                katexLink.rel = 'stylesheet';
                katexLink.href = 'https://cdn.jsdelivr.net/npm/katex@0.16.7/dist/katex.min.css';
                document.head.appendChild(katexLink);
                
                const katexScript = document.createElement('script');
                katexScript.src = 'https://cdn.jsdelivr.net/npm/katex@0.16.7/dist/katex.min.js';
                document.head.appendChild(katexScript);
                
                await new Promise(resolve => setTimeout(resolve, 500));
            }
            
            // Show preview
            previewFrame.srcdoc = htmlContent;
            preview.classList.add('show');
            
            // Options for html2pdf
            const opt = {
                margin: [0.5, 0.5, 0.5, 0.5],
                filename: `${selectedFile.name.replace('.zip', '').replace('.html', '')}.pdf`,
                image: { type: 'jpeg', quality: 0.98 },
                html2canvas: { scale: 2, useCORS: true, logging: false },
                jsPDF: { unit: 'in', format: 'a4', orientation: 'portrait' }
            };
            
            try {
                // Generate PDF
                const pdfBlob = await html2pdf().set(opt).from(container).outputPdf('blob');
                
                progressFill.style.width = '100%';
                progressText.textContent = 'Complete!';
                
                // Create download link
                const url = URL.createObjectURL(pdfBlob);
                const fileName = `${selectedFile.name.replace('.zip', '').replace('.html', '')}.pdf`;
                
                showResult(`
                    ✅ PDF created successfully!
                    <br><br>
                    <a href="${url}" download="${fileName}" class="download-link">
                        📥 Click here to download PDF
                    </a>
                `, 'success');
                
                setTimeout(() => {
                    progress.classList.remove('show');
                    convertBtn.disabled = false;
                }, 1000);
                
            } catch (error) {
                throw new Error(`PDF generation failed: ${error.message}`);
            }
        }

        function showResult(message, type) {
            resultDiv.innerHTML = message;
            resultDiv.className = `result ${type}`;
        }

        function hideResult() {
            resultDiv.className = 'result';
            resultDiv.style.display = 'none';
            resultDiv.innerHTML = '';
        }

        function hidePreview() {
            preview.classList.remove('show');
            previewFrame.srcdoc = '';
        }
    </script>
